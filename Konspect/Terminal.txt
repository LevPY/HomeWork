pwd - посмотреть где ты

mkdir name - создать папку

cd name - зайти в папку

ls -la - посмотреть файлы в текущей папке

touch file - создать пустой файл

cat > file - создать файл и редактировать его

cat file - просмотреть файл

cat >> file - редактировать файл

git status - проверить статус

git add file - добавить файл

git push - запушить внешний 

git commit - сделать сохранение

git clone - клонировать репозиторий 

git pull - запушить локальный
